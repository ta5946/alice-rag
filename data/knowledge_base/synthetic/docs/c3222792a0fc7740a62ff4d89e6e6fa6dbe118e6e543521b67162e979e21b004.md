#!/usr/bin/env bash

# ensure O2DPG and O2 are loaded
[ ! "${O2DPG_ROOT}" ] && echo "Error: O2DPG needs to be loaded" && exit 1

NBKGEVENTS=${NBKGEVENTS:-1}
NWORKERS=${NWORKERS:-8}
