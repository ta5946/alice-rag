#!/usr/bin/env bash
#
# This script demonstrates a hybrid generator simulation where multiple instances of the same generator
# are configured from a JSON file (hybridconfig_parallel.json) and run in parallel to enhance event production speed.

NEV=100
WORKERS=8

# Initiating the simulation using the Hybrid generator in parallel mode
${O2_ROOT}/bin/o2-sim --noGeant -j 1 --field ccdb --vertexMode kCCDB --run 300000 --configKeyValues "GeneratorHybrid.configFile=$PWD/hybridconfig_parallel.json;GeneratorHybrid.randomize=false;GeneratorHybrid.num_workers=${WORKERS}" -g hybrid -o genevents_parallel --timestamp 1546300800000 --seed 836302859 -n $NEV
