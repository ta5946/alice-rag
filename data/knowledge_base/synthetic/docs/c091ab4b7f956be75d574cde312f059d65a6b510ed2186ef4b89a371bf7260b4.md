#!/bin/bash
dplname=$1
while true; do
  proc=$(ps aux | grep "id ${dplname}" | grep -v "grep" | awk '{print $2}')
  if [ -n "${proc}" ]; then
    echo "FOUND ${proc} ... attaching gdb"
    gdb --pid ${proc}
    exit 0
  fi
  sleep 0.01
done
