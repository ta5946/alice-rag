#!/bin/bash

# Script designed to format JSON files

for file in $(find . | grep "\.json"); do
    case "$file" in
    *analyses_config.json*)
        continue
        ;;
    esac
    echo "Formatting $file"
    jq -S . $file >a.json
    mv a.json $file
done
